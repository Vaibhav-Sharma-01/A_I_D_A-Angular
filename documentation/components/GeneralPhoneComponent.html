import { Component, OnInit } from '@angular/core';

// AIDA Services
import { RegistriesService } from '@features/registries/service/registries.service';
import { ToastMessage } from '@shared/services/toast-message';

@Component({
  selector: 'app-general-phone',
  templateUrl: './general-phone.component.html',
  styleUrls: ['./general-phone.component.scss']
})
export class GeneralPhoneComponent implements OnInit {

  phone:string = '';
  phoneInput: string;
  isCalling: boolean = false;
  isPhoneValid: boolean = false;

  constructor(
    private registriesService: RegistriesService,
    private toast: ToastMessage
  ) { }

  ngOnInit(): void {
  }

  num(word: string) {
    this.isPhoneValid = false;
    this.phone += word;
    this.phoneInput = this.phone;
    if (this.phoneInput.match(/^[\+]?[(]?[0-9]{3}[)]?[-\s\.]?[0-9]{3}[-\s\.]?[0-9]{4,6}$/im)) {
      this.isPhoneValid = true;
    }
  }

  call() {
    if (this.isPhoneValid) {
      const data = {
        "url": "https://4a6658268240.ngrok.io/v1/api/call",
        "to": "+917042618988",
        "from": "+12566459370",
        "callTo": "+918791234693"
      };    
      this.registriesService.callPhone(data).subscribe(res => {
        this.isCalling = true;
        this.toast.open(`Calling ${this.phoneInput}...`, 'success');
      }, error => {
        this.toast.open(`Unable to make a call.`, 'error');
      })
    }
  }

  clearAll() {
    this.phoneInput = '';
    this.phone = '';
    this.isPhoneValid = false;
  }

  callEnd() {
    this.isCalling = false;
  }
}
